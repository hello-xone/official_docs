import ChangelogTimeline, { ChangelogEntry } from "@components/Changelog/Timeline";
import { Video } from "@components/Video";
import { Accordion } from "@components/Changelog/Accordion";
import RoadmapData from "../src/roadmap.json";

# Xone Roadmap

The [Xone](https://xone.org/) Roadmap outlines the long-term vision for our ecosystem — from infrastructure upgrades and application growth to governance and community building.

We will advance in phases to enhance network performance, enable cross-chain interoperability, implement decentralized governance, and expand developer incentives, driving [Xone](https://xone.org/) toward becoming a leading global Web3 infrastructure.

<ChangelogTimeline className="mt-8">
  {RoadmapData.entries.map((entry, idx) => (
    <ChangelogEntry
      key={idx}
      date={entry.date}
      version={entry.version}
      title={entry.title}
      tags={entry.tags}
    >
      <div className="space-y-4">
      {entry.blocks?.map((block, bIdx) => {
        if (block.type === "text") {
          const text = typeof block.text === "string" ? block.text : (block.content ?? "");
          const paragraphs = text.split(/\n{2,}/).filter(Boolean);
          return (
            <div key={bIdx}>
              {paragraphs.length > 0
                ? paragraphs.map((p, pi) => <p key={pi}>{p}</p>)
                : <p>{text}</p>}
            </div>
          );
        }
        if (block.type === "video") {
          return (
            <div key={bIdx} className="rounded-lg overflow-hidden border border-gray-200 dark:border-gray-800">
              <Video src={block.src} isMt6={Boolean(block.isMt6)} />
            </div>
          );
        }
        if (block.type === "accordion") {
          return (
            <Accordion key={bIdx}>
              {block.items?.map((item, iIdx) => (
                <Accordion.Item key={iIdx} title={item.title} defaultOpen={Boolean(item.defaultOpen)}>
                  {item.list ? (
                    <ul>
                      {item.list.map((li, lIdx) => (
                        <li key={lIdx}>
                          {li.label ? <strong>{li.label}</strong> : null}
                          {li.label && li.desc ? ": " : null}
                          {li.desc ?? null}
                        </li>
                      ))}
                    </ul>
                  ) : null}
                </Accordion.Item>
              ))}
            </Accordion>
          );
        }
        if (block.type === "list") {
          return (
            <ul key={bIdx}>
              {block.list?.map((li, lIdx) => (
                <li key={lIdx}>
                  {li.label ? <strong>{li.label}</strong> : null}
                  {li.label && li.desc ? " " : null}
                  {li.desc ?? null}
                </li>
              ))}
            </ul>
          );
        }
        return null;
      })}
      </div>
    </ChangelogEntry>
  ))}
</ChangelogTimeline>
