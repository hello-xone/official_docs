import BlogContent from '../../blog.mdx'
import { getAllBlogs } from '@lib/all-blogs'
import { extractRelevantTags } from '@lib/tags'
import { getAllPages } from 'nextra/context'

export async function getStaticPaths() {
  const allBlogs = getAllBlogs(getAllPages());
  const tags = extractRelevantTags(allBlogs);
  
  return {
    paths: tags.map(tagObj => ({
      params: { tag: tagObj.tag }
    })),
    fallback: true // Set to false if you want to generate all tags at build time
  };
}

export async function getStaticProps() {
  const allBlogs = getAllBlogs(getAllPages());
  const tags = extractRelevantTags(allBlogs);
  
  return {
    props: {
      initialArticles: allBlogs,
      initialTags: tags,
    },
    revalidate: 60,
  };
}

export default function TagPage(props) {
  return <BlogContent {...props} />
}
